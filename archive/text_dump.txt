```python
import numpy as np
from scipy.ndimage import gaussian_filter

class OcclusionSimulator:
    def __init__(self, img_size=512):

    def add_tree_cover(self, image, mask, n_trees=5):

        img_h, img_w = image.shape[:2]

            y, x = np.where(mask > 0)
            if len(y) == 0: continue
            idx = np.random.randint(len(y))
            cx, cy = x[idx], y[idx]
            
            # Random radius (5-20 meters equivalent)
            radius = np.random.randint(20, 80)
            color = (np.random.randint(30, 60), np.random.randint(80, 120), np.random.randint(30, 60))
            
            cv2.circle(occluded_img, (cx, cy), radius, color, -1)
            cv2.circle(tree_mask, (cx, cy), radius, 1, -1)
            
        # Apply blur to simulate canopy edges
        tree_mask = gaussian_filter(tree_mask, sigma=2)
        return occluded_img, tree_mask
    
    def add_building_shadows(self, image, angle_range=(30, 60)):
        """Add elongated shadows from buildings"""
        angle = np.random.uniform(*angle_range)
        length = np.random.randint(50, 150)
        shadow_mask = np.zeros(image.shape[:2], dtype=np.uint8)
        
        # Draw random rectangles as shadows
        for _ in range(3):
            x, y = np.random.randint(0, image.shape[1]), np.random.randint(0, image.shape[0])
```

### 3. Loss Function Engineering

Single losses fail on occluded boundaries. Use a **weighted combination**:

```python
class CombinedLoss(nn.Module):
    def __init__(self, alpha=0.4, beta=0.3, gamma=0.2, delta=0.1):
        super().__init__()
        self.alpha = alpha  # Dice
        self.beta = beta    # IoU
        self.gamma = gamma  # Boundary
        self.delta = delta  # Connectivity
        
    def dice_loss(self, pred, target, smooth=1.):
        intersection = (pred * target).sum()
        return 1 - (2. * intersection + smooth) / (pred.sum() + target.sum() + smooth)
    
    def boundary_loss(self, pred, target):
        """Penalize errors on road edges using Sobel gradient"""
        sobel_x = torch.tensor([[-1, 0, 1], [-2, 0, 2], [-1, 0, 1]], dtype=torch.float32).cuda()
        sobel_y = sobel_x.t()
        
        # Apply to target to get boundary map
        target_boundary = torch.abs(
            torch.nn.functional.conv2d(target, sobel_x.view(1,1,3,3), padding=1)
        ) + torch.abs(
            torch.nn.functional.conv2d(target, sobel_y.view(1,1,3,3), padding=1)
        )
        target_boundary = (target_boundary > 0).float()
        
        # Weight errors near boundaries higher
        weighted_bce = -(target_boundary * target * torch.log(pred + 1e-7) + 
                        target_boundary * (1-target) * torch.log(1-pred + 1e-7))
        return weighted_bce.mean()
    
    def connectivity_loss(self, pred, target):
        """Ensure skeletonized prediction matches target topology"""
        # Approximate using morphological operations in differentiable way
        # Or use pre-computed skeleton masks
        pred_skel = self.soft_skeletonize(pred)
        target_skel = self.soft_skeletonize(target)
        return self.dice_loss(pred_skel, target_skel)
    
    def soft_skeletonize(self, x, num_iterations=5):
        """Differentiable approximation of skeletonization"""
        for _ in range(num_iterations):
            # Erosion simulation via min pooling
            x = torch.nn.functional.max_pool2d(1-x, kernel_size=3, stride=1, padding=1)
            x = 1 - x
        return x
    
    def forward(self, pred, target):
        dice = self.dice_loss(pred, target)
        iou = 1 - (pred * target).sum() / ((pred + target) - (pred * target)).sum().clamp(min=1e-7)
        bnd = self.boundary_loss(pred, target)
        conn = self.connectivity_loss(pred, target)
        
        return self.alpha * dice + self.beta * iou + self.gamma * bnd + self.delta * conn
```

**Training Loop with Hard Negative Mining:**

```python
# During training, identify hard examples (occluded roads with high loss)
# and resample them in the next batch
def train_step(model, batch, optimizer, criterion):
    images, masks = batch
    images, masks = images.cuda(), masks.cuda()
    
    # Forward
    outputs = model(images)
    loss = criterion(torch.sigmoid(outputs), masks)
    
    # Identify hard negatives (high loss pixels in occluded regions)
    with torch.no_grad():
        pixel_loss = torch.nn.functional.binary_cross_entropy(
            outputs, masks, reduction='none'
        )
        # Store hard sample indices for next epoch
        hard_indices = torch.where(pixel_loss > 0.5)  
    
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    return loss.item()
```

```python
def mask_to_graph(mask, transform, threshold=0.5):
    """
    Convert binary mask to NetworkX graph with geo-coordinates
    """
    # Binarize and skeletonize
    binary = (mask > threshold).astype(np.uint8)
    skele = skeletonize(binary).astype(np.uint8)
    
    # Extract nodes (junctions and endpoints)
    from skimage.morphology import binary_dilation
    from scipy.ndimage import convolve
    
    # Hit-or-miss for finding junctions (3+ connections)
    kernel = np.array([[1, 1, 1],
                       [1, 0, 1],
                       [1, 1, 1]])
    neighbors = convolve(skele, kernel, mode='constant', cval=0)
    junctions = (skele == 1) & (neighbors >= 3)
    endpoints = (skele == 1) & (neighbors == 1)
    
    # Create graph
    G = nx.Graph()
    
    # Vectorize skeleton to edges using rasterio features
    shapes = rasterio.features.shapes(skele, mask=skele)
    for geom, val in shapes:
        if val == 1:
            coords = list(geom['coordinates'])
            # Simplify line
            line = LineString(coords).simplify(1.0)
            
            # Add edges between nodes
            start, end = line.coords[0], line.coords[-1]
            start_idx = f"{start[0]:.6f},{start[1]:.6f}"
            end_idx = f"{end[0]:.6f},{end[1]:.6f}"
            
            # Calculate length in meters using transform
            length = line.length * abs(transform[0])  # Approximate
            
            G.add_edge(start_idx, end_idx, 
                      weight=length, 
                      geometry=line,
                      coordinates=coords)
            
    return G, junctions, endpoints
```

### 2. Topological Healing (The MST Bridge)

When occlusion breaks a road, you need to heal it logically:

```python
from scipy.spatial import KDTree
import numpy as np

def heal_graph_topology(G, endpoints, max_gap=50, angle_threshold=30):
    """
    Connects broken road segments using geometric logic
    """
    # Extract endpoint coordinates
    endpoint_nodes = [n for n in G.nodes() if G.degree(n) == 1]
    if len(endpoint_nodes) < 2:
        return G
    
    coords = np.array([tuple(map(float, n.split(','))) for n in endpoint_nodes])
    tree = KDTree(coords)
    
    # Find pairs within max_gap
    pairs = tree.query_pairs(r=max_gap, output_type='ndarray')
    
    healing_candidates = []
    for i, j in pairs:
        node_a, node_b = endpoint_nodes[i], endpoint_nodes[j]
        coord_a, coord_b = coords[i], coords[j]
        
        # Check angular alignment (roads should continue in similar direction)
        vec_ab = coord_b - coord_a
        
        # Get tangent vectors at endpoints from existing edges
        tan_a = get_tangent(G, node_a)
        tan_b = get_tangent(G, node_b)
        
        # Calculate angles
        angle_a = angle_between(tan_a, vec_ab)
        angle_b = angle_between(tan_b, -vec_ab)  # Reverse for node_b
        
        if angle_a < angle_threshold and angle_b < angle_threshold:
            healing_candidates.append((node_a, node_b, np.linalg.norm(vec_ab)))
    
    # Use MST to select optimal healing edges without creating cycles unnecessarily
    if healing_candidates:
        healing_candidates.sort(key=lambda x: x[2])  # Sort by distance
        
        # Union-Find to prevent cycles
        parent = {n: n for n in G.nodes()}
        def find(x): return x if parent[x] == x else find(parent[x])
        def union(x, y): parent[find(x)] = find(y)
        
        for node_a, node_b, dist in healing_candidates:
            if find(node_a) != find(node_b):
                # Check if this connects two components or closes a gap
                G.add_edge(node_a, node_b, weight=dist, healed=True)
                union(node_a, node_b)
                
    return G

def get_tangent(G, node):
    """Extract direction of road at endpoint"""
    neighbor = list(G.neighbors(node))[0]
    coord_node = np.array([float(x) for x in node.split(',')])
    coord_neighbor = np.array([float(x) for x in neighbor.split(',')])
    return coord_neighbor - coord_node

def angle_between(v1, v2):
    """Calculate angle between vectors in degrees"""
    cos_angle = np.dot(v1, v2) / (np.linalg.norm(v1) * np.linalg.norm(v2) + 1e-7)
    return np.degrees(np.arccos(np.clip(cos_angle, -1.0, 1.0)))
```

## Phase III: Network Analysis & Stress Testing

Now analyze vulnerability using graph theory.

```python
import networkx as nx

class ResilienceAnalyzer:
    def __init__(self, G):
        self.G = G
        self.baseline_efficiency = self._calculate_efficiency()
        self.centrality = nx.betweenness_centrality(G, weight='weight', normalized=True)
        
    def _calculate_efficiency(self):
        """Global efficiency: average inverse shortest path length"""
        n = len(self.G)
        if n < 2:
            return 0
        
        eff = 0
        for node in self.G:
            # Single-source shortest path
            lengths = nx.single_source_dijkstra_path_length(self.G, node, weight='weight')
            for target, length in lengths.items():
                if node != target:
                    eff += 1.0 / length
        return eff / (n * (n - 1))
    
    def calculate_resilience_index(self, removed_nodes):
        """
        R = APL_baseline / APL_perturbed
        Lower R indicates higher vulnerability (APL increases when critical nodes removed)
        """
        G_perturbed = self.G.copy()
        G_perturbed.remove_nodes_from(removed_nodes)
        
        if not nx.is_connected(G_perturbed):
            # Handle disconnected components - efficiency approaches 0
            return 0.0
        
        perturbed_efficiency = self._calculate_efficiency(G_perturbed)
        
        # Convert to average path length (approximate)
        # Efficiency ≈ 1/<L>, so <L> ≈ 1/Efficiency
        apl_baseline = 1.0 / (self.baseline_efficiency + 1e-7)
        apl_perturbed = 1.0 / (perturbed_efficiency + 1e-7)
        
        resilience_index = apl_baseline / apl_perturbed
        return resilience_index
    
    def stress_test_cascading_failure(self, top_k_percent=5):
        """
        Simulate cascading failures by iteratively removing highest centrality nodes
        """
        nodes_by_centrality = sorted(self.centrality.items(), key=lambda x: x[1], reverse=True)
        num_remove = int(len(self.G) * top_k_percent / 100)
        
        results = []
        removed_set = set()
        
        for i in range(num_remove):
            node_to_remove = nodes_by_centrality[i][0]
            removed_set.add(node_to_remove)
            
            ri = self.calculate_resilience_index(removed_set)
            results.append({
                'step': i,
                'node_removed': node_to_remove,
                'centrality': nodes_by_centrality[i][1],
                'resilience_index': ri,
                'nodes_disconnected': len(self.G) - len(removed_set) - len(
                    max(nx.connected_components(self.G.subgraph(set(self.G.nodes()) - removed_set)), key=len)
                    if len(removed_set) < len(self.G) else []
                )
            })
            
        return results
    
    def identify_critical_corridors(self):
        """Return edges that appear in >80% of shortest paths between random pairs"""
        edge_counts = {}
        nodes = list(self.G.nodes())
        
        # Sample random pairs (for large graphs)
        import random
        samples = min(1000, len(nodes) * (len(nodes) - 1) // 2)
        
        for _ in range(samples):
            u, v = random.sample(nodes, 2)
            try:
                path = nx.shortest_path(self.G, u, v, weight='weight')
                for i in range(len(path) - 1):
                    edge = tuple(sorted([path[i], path[i+1]]))
                    edge_counts[edge] = edge_counts.get(edge, 0) + 1
            except nx.NetworkXNoPath:
                continue
                
        threshold = samples * 0.8
        critical_edges = [edge for edge, count in edge_counts.items() if count > threshold]
        return critical_edges
```

The prompt specifically asks for **Transformer-based architectures** to handle long-range dependencies (knowing a road continues even if 50 pixels are hidden).

*   **Recommendation:** Use **Swin-Unet** or **SegFormer**. These combine the local feature extraction of CNNs with the global attention of Transformers.
*   **Why?** A standard U-Net might break the road line at a shadow. A Transformer attends to the road segments before and after the shadow and infers the connection.
*   **Library:** `segmentation_models_pytorch` (SMP) makes this easy.

```python
import segmentation_models_pytorch as smp

# Example: Swin-Unet (Transformer Encoder + U-Net Decoder)
model = smp.Unet(
    encoder_name="swin_b",  # Transformer backbone
    encoder_weights="imagenet",
    in_channels=3,
    classes=1,
    activation='sigmoid'
)
```

### 3. Loss Function Engineering
Roads are thin (class imbalance). Standard CrossEntropy fails. You need a combination loss.

*   **Dice Loss:** Handles class imbalance (focuses on overlap).
*   **BCE Loss:** Standard binary classification error.
*   **Boundary Loss (Optional but recommended):** Penalizes errors on the edges of the road to keep lines thin and precise.
*   **Implementation:**
    ```python
    def combined_loss(pred, target):
        dice = smp.losses.DiceLoss(mode='binary')(pred, target)
        bce = smp.losses.SoftBCEWithLogitsLoss()(pred, target)
        return 0.7 * dice + 0.3 * bce
    ```

### 4. Training Loop Strategy
*   **Optimizer:** AdamW (better for Transformers).
*   **Scheduler:** Cosine Annealing (helps converge to a better minimum).
*   **Validation:** Monitor **IoU (Intersection over Union)** and **Connectivity Score**.
    *   *Connectivity Score:* A custom metric that checks if the predicted road is one continuous line or broken fragments.

# Phase II: Graph Skeletonization & Healing
**Goal:** Convert the binary road mask (Phase I output) into a navigable Graph (Nodes & Edges).

### 1. Thinning (Skeletonization)
The model outputs a thick mask (e.g., 10 pixels wide). You need a 1-pixel centerline.
*   **Tool:** `skimage.morphology.skeletonize`.
*   **Process:**
    1.  Threshold the model prediction (e.g., > 0.5 = Road).
    2.  Apply `skeletonize` to get 1-pixel width lines.

### 2. Vectorization (Nodes & Edges)
Convert the pixel skeleton into a Graph.
*   **Nodes:** Identify **Junctions** (pixels with > 2 neighbors) and **Endpoints** (pixels with 1 neighbor).
*   **Edges:** The path of pixels connecting two nodes.
*   **Tool:** `networkx` or custom scanning algorithm.

### 3. Topological Healing (Bridging Gaps)
Occlusions might still cause small breaks in the skeleton. You need to "heal" them logically.
*   **Algorithm:**
    1.  Find all **Endpoints** in the graph.
    2.  For every pair of endpoints $(A, B)$:
        *   Calculate Euclidean Distance $d$.
        *   Calculate Angle of approach $\theta$.
    3.  **Condition:** If $d < \text{threshold}$ (e.g., 15 pixels) AND angles align (they are facing each other):
        *   Draw a line between A and B.
        *   Add this edge to the graph.
*   **Advanced:** Use **Minimum Spanning Tree (MST)** logic on the endpoints to connect disjoint components with the minimum amount of new "road."

# Phase III: Network Analysis & Stress Testing
**Goal:** Mathematically prove which roads are critical and how the network fails.

### 1. Centrality Calculation
*   **Metric:** **Betweenness Centrality**.
*   **Meaning:** How often does a node appear on the shortest path between all other nodes?
*   **Interpretation:** High score = Critical Bottleneck (e.g., a specific bridge). If this fails, traffic cannot reroute easily.
*   **Code:**
    ```python
    import networkx as nx
    centrality = nx.betweenness_centrality(G)
    ```

### 2. Node Ablation Simulation (Stress Test)
Simulate a disaster (flood/earthquake) by removing critical nodes.
*   **Step 1:** Calculate Baseline Global Efficiency ($E_{base}$).
    *   $E = \frac{1}{N(N-1)} \sum \frac{1}{d_{ij}}$ (Average inverse shortest path).
*   **Step 2:** Remove the top 5% highest centrality nodes.
*   **Step 3:** Calculate New Efficiency ($E_{new}$).
*   **Step 4:** Calculate **Resilience Index (R)**.
    *   $R = \frac{E_{new}}{E_{base}}$
    *   If $R$ is close to 1, the network is robust. If $R$ drops significantly, the network is fragile.

### Pro-Tip for the Training Phase
Since training Transformers is computationally heavy, if you are limited on GPU resources during the hackathon:
1.  Use a **pre-trained encoder** (ImageNet weights).
2.  Freeze the encoder for the first few epochs, then unfreeze.
3.  Use **Mixed Precision Training (AMP)** to save memory and speed up training by 2x.
    ```python
    from torch.cuda.amp import autocast, GradScaler
    scaler = GradScaler()
    with autocast():
        output = model(image)
        loss = criterion(output, mask)
    scaler.scale(loss).backward()
    ```
